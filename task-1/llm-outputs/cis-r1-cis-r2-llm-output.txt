import json

def analyze_documents(pdf_inputs_path):
    """
    Analyzes two documents to extract key data elements and structure them in a nested dictionary.

    Args:
        pdf_inputs_path (str): The path to the PDF files.

    Returns:
        dict: A nested dictionary containing the extracted data elements.
    """

    documents = {
        "cis_r1": {},
        "cis_r2": {}
    }

    try:
        with open(pdf_inputs_path, 'r') as f:
            doc1 = json.load(f)
            doc2 = json.load(f)

        # Analyze cis_r1
        if "cis_r1" in doc1:
            cis_r1_data = {
                "title": doc1["cis_r1"]["title"],
                "requirements": [
                    "human-readable",
                    "descriptive",
                    "strategic",
                    "technical"
                ]
            }
            documents["cis_r1"] = cis_r1_data
        else:
            documents["cis_r1"] = {"title": "No CIS R1 document found", "requirements": []}

        # Analyze cis_r2
        if "cis_r2" in doc2:
            cis_r2_data = {
                "title": "No CIS R2 document found",
                "requirements": [
                    "technical",
                    "operational",
                    "automation"
                ]
            }
            documents["cis_r2"] = cis_r2_data
        else:
            documents["cis_r2"] = {"title": "No CIS R2 document found", "requirements": []}


    except FileNotFoundError:
        print(f"Error: File not found at {pdf_inputs_path}")
        return None
    except json.JSONDecodeError:
        print(f"Error: Invalid JSON format in {pdf_inputs_path}")
        return None

    return documents
